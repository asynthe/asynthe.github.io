install

  1. Download EAF $ git clone --depth=1 -b master https://github.com/emacs-eaf/emacs-application-framework.git ~/.emacs.d/site-lisp/emacs-application-framework/

  2. Install/Update EAF applications and dependencies (see NixOS header if using NixOS) $ cd ~/.emacs.d/site-lisp/emacs-application-framework $ chmod +x ./install-eaf.py $ ./install-eaf.py

    If using Wayland: check the github page If using NixOS: if pacakges are not recognized, you can install with flags $ ./install-eaf.py --ignore-sys-deps

  3. Load the EAF core In config.el, add the next lines 

    (add-to-list 'load-path "~/.emacs.d/site-lisp/emacs-application-framework/")
(require 'eaf)

    Or, if using use-package, add the next configuration: 

    (use-package eaf
  :load-path "~/.emacs.d/site-lisp/emacs-application-framework"
  :custom
  ; See https://github.com/emacs-eaf/emacs-application-framework/wiki/Customization
  (eaf-browser-continue-where-left-off t)
  (eaf-browser-enable-adblocker t)
  (browse-url-browser-function 'eaf-open-browser)
  :config
  (defalias 'browse-web #'eaf-open-browser)
  (eaf-bind-key scroll_up "C-n" eaf-pdf-viewer-keybinding)
  (eaf-bind-key scroll_down "C-p" eaf-pdf-viewer-keybinding)
  (eaf-bind-key take_photo "p" eaf-camera-keybinding)
  (eaf-bind-key nil "M-q" eaf-browser-keybinding)) ;; unbind, see more in the Wiki

  4. Load Apps 

    (require 'eaf-browser)
(require 'eaf-pdf-viewer)

install in NixOS

add the next to your configuration.nix 

let
  python-packages = ps: with ps; [
    # For Emacs EAF
    pandas
    requests
    pyqt6 sip qtpy epc lxml pyqt6-webengine # for eaf
    qrcode # eaf-file-browser
    pysocks # eaf-browser
    pymupdf # eaf-pdf-viewer
    pypinyin # eaf-file-manager
    psutil # eaf-system-monitor
    retry # eaf-markdown-previewer
    markdown
  ];

in {
  environment.systemPackages = with pkgs; [
    (pkgs.python3.withPackages python-packages)
  ];
}

KEYBINDS

VANILLA EMACS (just in case)

commands M-x text-increse -> increase font size M-x find-file -> look for and open a file M-x load-file -> load a file (useful for reloading emacs) eshell -> a terminal to do stuff

keybindings C-x C-f -> find-file

C-x C-s -> save-buffer C-x k -> kill-buffer ESC -> close annoying minibuffers just pressing that

M-w -> copy text C-y -> paste text C-w -> cut text

C-/ or C-x u or C-_ -> undo C-g C-_ -> redo (redo its just undo, just do undo again)

window splits C-x 2 -> split window horizontal C-x 3 -> split window vertically window splits stuff https://stackoverflow.com/questions/4987760/how-to-change-size-of-split-screen-emacs-windows
