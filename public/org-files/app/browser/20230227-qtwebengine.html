install

binary package in gentoo

  • it can be installed with emerge -G qtwebengine by using on of these two options.

in /etc/portage/make.conf 

PORTAGE_BINHOST="https://gentoo.osuosl.org/experimental/amd64/binpkg/default/linux/17.1/x86-64/"

in /etc/portage/binrepos.conf 

[gentoo-binpkg]
priority = 9999
sync-uri = https://gentoo.osuosl.org/experimental/amd64/binpkg/default/linux/17.1/x86-64/
