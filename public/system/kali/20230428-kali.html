CHROOT

on your linux install or live usb, do as follows # mount /dev/<root-partition> /mnt # mount /dev/<boot-partition> /mnt/boot/efi

mounting some stuff # mount --bind /dev /mnt/dev && # mount --bind /dev/pts /mnt/dev/pts && # mount --bind /proc /mnt/proc && # mount --bind /sys /mnt/sys

the chrooting # sudo chroot /mnt /bin/bash -> /bin/bash to specify a shell, as in my case i use [dash]

the installing # grub-install --target=x86_64-efi --efi-directory=/boot/efi --removable

unmounting and saving $ exit && # umount /mnt/sys && # umount /mnt/proc && # umount /mnt/dev/pts && # umount /mnt/dev && # umount /mnt
